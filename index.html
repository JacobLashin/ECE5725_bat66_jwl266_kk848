            def generate_features(coords):
                relative = []
                x0, y0 = coords[0]
                x1, y1 = coords[1]
                d = np.sqrt((x1-x0)*(x1-x0) + (y1-y0)*(y1-y0))
                for i in range(20):
                    x, y = coords[i+1]
                    relative.append((x-x0)/d)
                    relative.append((y-y0)/d)
                return relative
